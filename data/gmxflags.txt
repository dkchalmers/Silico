#? The list of standard gromacs flags

#. Gromacs settings:
#; -g__exe_<string>    Gromacs executable (default gmx)
#; -g__nt_<int>        Gromacs -nt flag
#; -g__maxwarn_<num>   Gromacs grompp maximum number of warnings
#; -g__pin             Gromacs pin threads flag
#; -g__pinstride_<int> Gromacs pinstride for use with gmx-pin
#; -g__pinoffset_<arr> Space delimited array of pinoffset values
#; -g__f               Additional gromacs flags

#. Gromacs local job options:
#; -h_<hostlist>      Host list for local jobs. Space separated string. Use quotes on the command line.
#; -maxj_<int>        Maximum number of GPU jobs(gmx, desmond, etc) allowed per GPU

#. SLURM job options:
#; -s__qos_<string>   SLURM quality of services 
#; -s__time_<string>  SLURM maximum job run time (hrs:mins:secs)
#; -s__nnode_<num>    SLURM number of nodes to request (default 1)
#; -s__ncpu_<num>     SLURM number cpus
#; -s__ngpu_<num>     SLURM number of GPUs
#; -s__part_<string>  SLURM partition
